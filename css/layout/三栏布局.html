# CSS 三栏布局

create by yuzhoufen on 2019-09-19 15:45

  • html Demo
<div class="container">
    <!-- 优先渲染 -->
    <div class="center">
    center
    </div>
    <div class="left">
    left
    </div>
    <div class="right">
    right
    </div>
</div>
  • flex 三栏布局
.container {
    display: flex;
    height: 200px;
}
.center {
    background-color: red;
    width: 100%;
    order: 2;
}
.left {
    background-color: green;
    width: 200px;
    flex-shrink: 0;
    order: 1;
}
.right {
    background-color: blue;
    width: 200px;
    flex-shrink: 0;
    order: 3;
}
  • table-cell三栏布局
.container {
    overflow: hidden;
    position: relative;
}
.container > div {
    display: table-cell;
    height: 100%;
}
.center {
    margin: 0 200px;
    width: 100%;
    background: red;
}
.left {
    width: 200px;
    min-width: 200px;
    background-color: green;
}
.right {
    width: 200px;
    min-width: 200px;
    background-color: blue;
}
  • grid网格布局
 .container {
    display: grid;
    width: 100%;
    grid-template-rows: 100px;
    grid-template-columns: 200px auto 200px;
}
.center {
    background: red;
}
.left {
    background-color: green;
}
.right {
    background-color: blue;
}
